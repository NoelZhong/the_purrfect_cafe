<script>

  let score = 15; 

  let cats = [];

  let customers = [];

  let currentCustomer = null;

  let upgrades = {

    faster_service: false,

    more_cats: false,

    better_recipes: false

  };

  let gameTimer = null;

  let timeRemainingSeconds = 0;


  let ingredientsStock = {

    Coffee: 0,

    Tea: 0,

    Snack: 0

  };

  const ingredientPrices = {

    Coffee: 5,

    Tea: 4,

    Snack: 6

  };


  const catNames = ["Whiskers", "Mittens", "Shadow", "Luna", "Simba", "Oliver", "Chloe"];

  const customerNames = ["Alice", "Bob", "Charlie", "Diana", "Ethan", "Fiona"];


  // Start button handler

document.getElementById('startButton').onclick = () => {

  startGame();

};


function startGame() {

  const timingOption = document.querySelector('input[name="timing"]:checked').value;

  let minutes = 0;

  if (timingOption === 'minutes') {

    minutes = parseInt(document.getElementById('minutesInput').value);

    if (isNaN(minutes) || minutes < 1) {

      alert("Please enter a valid number of minutes (at least 1).");

      return;

    }

  }

  const difficulty = document.getElementById('difficultySelect').value;

  if (difficulty === 'easy') {

    score = 30;

  } else if (difficulty === 'medium') {

    score = 15;

  } else if (difficulty === 'hard') {

    score = 10;

  }

  document.getElementById('score').innerText = 'Score: ' + score;

  document.getElementById('titleScreen').style.display = 'none';

  document.getElementById('mainGame').style.display = 'block';


  if (timingOption === 'minutes') {

    timeRemainingSeconds = minutes * 60;

    updateTimerDisplay();

    window.gameInterval = setInterval(() => {

      timeRemainingSeconds--;

      if (timeRemainingSeconds <= 0) {

        clearInterval(window.gameInterval);

        endGame();

      } else {

        updateTimerDisplay();

      }

    }, 1000);

  } else {

    document.getElementById('timerDisplay').innerText = '';

  }


  displayCats();

  displayIngredients();

  updateScore();

}


function updateTimerDisplay() {

  const minutes = Math.floor(timeRemainingSeconds / 60);

  const seconds = timeRemainingSeconds % 60;

  document.getElementById('timerDisplay').innerText = `Time Remaining: ${minutes}:${seconds.toString().padStart(2, '0')}`;

}


function endGame() {

  alert("Time's up! Game over. Your final score: " + score);

  document.querySelectorAll('button').forEach(btn => btn.disabled = true);

  document.getElementById('feedback').innerText = 'Game Over! Final Score: ' + score;

}


// Core gameplay functions

function adoptCat() {

  if (cats.length >= 8 && !upgrades.more_cats) {

    alert("You have enough cats! (or buy upgrade to adopt more)");

    return;

  }

  const cost = 25;

  if (score < cost) {

    alert("Not enough points to adopt a cat! Need " + cost + " points.");

    return;

  }

  const name = catNames[Math.floor(Math.random() * catNames.length)];

  if (cats.includes(name)) {

    alert("This cat is already adopted!");

    return;

  }

  cats.push(name);

  score -= cost;

  displayCats();

  updateScore();

}


function displayCats() {

  const catsDiv = document.getElementById('cats');

  catsDiv.innerHTML = '';

  cats.forEach(cat => {

    const catDiv = document.createElement('div');

    catDiv.className = 'cat';

    catDiv.innerText = cat;

    catDiv.onclick = () => interactWithCat(cat);

    catsDiv.appendChild(catDiv);

  });

}


function addCustomer() {

  if (customers.length >= 5) {

    alert("Too many customers waiting!");

    return;

  }

  const name = customerNames[Math.floor(Math.random() * customerNames.length)];

  if (customers.includes(name)) {

    alert("This customer is already waiting!");

    return;

  }

  customers.push(name);

  displayCustomers();

}


function displayCustomers() {

  const customersDiv = document.getElementById('customers');

  customersDiv.innerHTML = '';

  customers.forEach(cust => {

    const custDiv = document.createElement('div');

    custDiv.className = 'customer';

    custDiv.innerText = cust;

    customersDiv.appendChild(custDiv);

  });

}


function serveCustomer() {

  if (customers.length === 0) {

    alert("No customers to serve!");

    return;

  }

  currentCustomer = customers.shift();

  document.getElementById('currentCustomer').innerHTML = 'Serving: ' + currentCustomer;

  displayCustomers();

  document.getElementById('feedback').innerText = 'Choose a recipe to serve!';

}


function prepareRecipe(recipe) {

  if (!currentCustomer) {

    alert("No customer is being served!");

    return;

  }

  if (ingredientsStock[recipe] <= 0) {

    alert("Need to buy ingredients for " + recipe + " first!");

    return;

  }

  let successChance = 0.7;

  if (upgrades.better_recipes) successChance = 0.9;

  const success = Math.random() < successChance;

  if (success) {

    score += 20;

    document.getElementById('feedback').innerText = currentCustomer + " loved the " + recipe + "! +20 points.";

  } else {

    score -= 10;

    document.getElementById('feedback').innerText = currentCustomer + " didn't like the " + recipe + ". -10 points.";

  }

  ingredientsStock[recipe]--;

  updateScore();

  currentCustomer = null;

  document.getElementById('currentCustomer').innerHTML = '';

  displayIngredients();

}


function buyIngredient(recipe) {

  const price = ingredientPrices[recipe];

  if (score < price) {

    alert("Not enough points to buy " + recipe + " ingredients!");

    return;

  }

  score -= price;

  ingredientsStock[recipe]++;

  updateScore();

  displayIngredients();

}


function displayIngredients() {

  const ingDiv = document.getElementById('ingredients');

  ingDiv.innerHTML = '';

  for (let recipe in ingredientsStock) {

    ingDiv.innerHTML += `<div>${recipe}: ${ingredientsStock[recipe]} 

      <button onclick="buyIngredient('${recipe}')">Buy (+1, ${ingredientPrices[recipe]} pts)</button></div>`;

  }

}


function buyUpgrade(upgrade) {

  if (score < 50 && upgrade !== 'more_cats') {

    alert("Not enough points for this upgrade!");

    return;

  }

  switch (upgrade) {

    case 'faster_service':

      if (!upgrades.faster_service) {

        upgrades.faster_service = true;

        score -= 50;

        alert("Faster Service unlocked!");

      } else {

        alert("Already purchased!");

      }

      break;

    case 'more_cats':

      if (!upgrades.more_cats) {

        if (score >= 100) {

          upgrades.more_cats = true;

          score -= 100;

          alert("More Cats unlocked! You can adopt more cats.");

        } else {

          alert("Need 100 points for this upgrade.");

        }

      } else {

        alert("Already purchased!");

      }

      break;

    case 'better_recipes':

      if (!upgrades.better_recipes) {

        score -= 75;

        upgrades.better_recipes = true;

        alert("Better Recipes unlocked! Customers are happier.");

      } else {

        alert("Already purchased!");

      }

      break;

  }

  updateScore();

}


function interactWithCat(cat) {

  const feedbackDiv = document.getElementById('feedback');

  feedbackDiv.innerText = `You played with ${cat}! They purr happily. +5 points!`;

  score += 5;

  updateScore();

}


function playWithCat() {

  if (cats.length === 0) {

    alert("No cats to play with!");

    return;

  }

  const cat = cats[Math.floor(Math.random() * cats.length)];

  interactWithCat(cat);

}


function updateScore() {

  document.getElementById('score').innerText = 'Score: ' + score;

}


// Timer functions

function updateTimerDisplay() {

  const minutes = Math.floor(timeRemainingSeconds / 60);

  const seconds = timeRemainingSeconds % 60;

}
